    <div id="app" class="w-full max-w-lg bg-white shadow-2xl rounded-xl overflow-hidden">
        
        <!-- Header -->
        <header class="p-6 bg-indigo-600 text-white rounded-t-xl">
            <h1 class="text-3xl font-extrabold">Funny Riddles and Comedy</h1>
            <p class="text-indigo-200 mt-1 text-sm">Fuel your brain with humor, challenges, and absurd questions.</p>
        </header>

        <!-- Navigation Tabs (Optimized for mobile taps) -->
        <nav class="flex justify-around p-3 bg-indigo-700 shadow-lg">
            <button onclick="showSection('brainstorm')" id="tab-brainstorm" class="tab-button bg-indigo-500 text-white flex-1 py-3 px-1 rounded-lg font-semibold text-xs sm:text-sm transition duration-200 hover:bg-indigo-400 focus:outline-none focus:ring-2 focus:ring-indigo-300 mx-1">
                Brainstorming Fun
            </button>
            <button onclick="showSection('jokes')" id="tab-jokes" class="tab-button text-indigo-100 flex-1 py-3 px-1 rounded-lg font-semibold text-xs sm:text-sm transition duration-200 hover:bg-indigo-500 focus:outline-none focus:ring-2 focus:ring-indigo-300 mx-1">
                Jokes & Humor
            </button>
            <button onclick="showSection('riddles')" id="tab-riddles" class="tab-button text-indigo-100 flex-1 py-3 px-1 rounded-lg font-semibold text-xs sm:text-sm transition duration-200 hover:bg-indigo-500 focus:outline-none focus:ring-2 focus:ring-indigo-300 mx-1">
                Riddles & Puzzles
            </button>
        </nav>

        <!-- Content Area -->
        <main class="p-4 content-area h-[60vh] overflow-y-auto">

            <!-- 1. Brainstorming Questions Section -->
            <section id="brainstorm-section" class="content-section space-y-4">
                <h2 class="text-2xl font-bold text-gray-800 border-b pb-2 mb-4">Absurd Brainstorming Questions</h2>
                <div id="brainstorm-list" class="space-y-3">
                    <!-- Questions are generated here -->
                </div>
            </section>

            <!-- 2. Jokes Section -->
            <section id="jokes-section" class="content-section space-y-4 hidden">
                <h2 class="text-2xl font-bold text-gray-800 border-b pb-2 mb-4">Simple, Funny Jokes</h2>
                <div id="jokes-list" class="space-y-4">
                    <!-- Jokes are generated here -->
                </div>
            </section>

            <!-- 3. Riddles & Puzzles Section (Game) -->
            <section id="riddles-section" class="content-section space-y-4 hidden">
                <h2 class="text-2xl font-bold text-gray-800 border-b pb-2 mb-4">Mind Games: Solve the Riddle</h2>
                <div class="bg-yellow-100 border-l-4 border-yellow-500 text-yellow-800 p-3 rounded-lg mb-4">
                    <p class="font-medium">Can you guess the answer? Click "Reveal" when you're ready!</p>
                </div>
                <div id="riddle-container" class="space-y-5">
                    <!-- Riddles are generated here -->
                </div>
            </section>

    <script>
        // --- Data Definitions ---

        const brainstormingQuestions = [
            "If you could combine any two animals to make the ultimate pet, what would they be and what would you call it?",
            "What common phrase or idiom would be terrifying if taken literally?",
            "If all vegetables suddenly gained the ability to talk, which one would be biggest jerk?",
            "What's the most useless superpower you can think of?",
            "If humanity had a warning label, what would it say?",
            "What is something that is always better in theory than in practice?",
            "You are now the CEO of a company that only sells things that glow. What is your best-selling product?",
            "If you could replace money with any object, what would be the new global currency?",
            "What cartoon character would be the absolute worst roommate, and why?",
            "If your car could talk, what's the most embarrassing thing it would reveal about you?"
        ];

        // Simple, easy-to-understand jokes
        const jokes = [
            { type: 'setup', setup: "What's brown and sticky?", punchline: "A stick." },
            { type: 'knockknock', setup: "Knock, knock. Who's there? Interrupting cow.", punchline: "MOO!" },
            { type: 'setup', setup: "Why was six afraid of seven?", punchline: "Because seven ate nine!" },
            { type: 'setup', setup: "What do you call a sleeping bull?", punchline: "A bulldozer." },
            { type: 'setup', setup: "What do you call a lazy kangaroo?", punchline: "Pouch potato." },
            { type: 'setup', setup: "Why did the banana put on sunscreen?", punchline: "Because it was peeling!" },
            { type: 'setup', setup: "What do you call an angry carrot?", punchline: "A steamed vegetable." },
            { type: 'setup', setup: "What is a pirate's favorite letter?", punchline: "The C." }
        ];

        const riddlesAndPuzzles = [
            { riddle: "I have cities, but no houses. I have mountains, but no trees. I have water, but no fish. What am I?", answer: "A map" },
            { riddle: "What has an eye but cannot see?", answer: "A needle" },
            { riddle: "What is full of holes but still holds water?", answer: "A sponge" },
            { riddle: "What is always coming, but never arrives?", answer: "Tomorrow" },
            { riddle: "What question can you never answer yes to?", answer: "Are you asleep yet?" },
            { riddle: "Feed me, and I live. Give me a drink, and I die. What am I?", answer: "Fire" },
            { riddle: "What word is spelled incorrectly in every dictionary?", answer: "Incorrectly" }
        ];

        // --- Utility Functions ---

        /**
         * Switches the visible content section and updates the navigation buttons.
         * @param {string} sectionId The ID of the section to show (e.g., 'brainstorm', 'jokes', 'riddles').
         */
        function showSection(sectionId) {
            // Hide all content sections
            document.querySelectorAll('.content-section').forEach(section => {
                section.classList.add('hidden');
            });

            // Deactivate all tabs
            document.querySelectorAll('.tab-button').forEach(button => {
                button.classList.remove('bg-indigo-500', 'text-white');
                button.classList.add('text-indigo-100', 'hover:bg-indigo-500');
            });

            // Show the requested section
            document.getElementById(`${sectionId}-section`).classList.remove('hidden');

            // Activate the corresponding tab
            const activeTab = document.getElementById(`tab-${sectionId}`);
            activeTab.classList.add('bg-indigo-500', 'text-white');
            activeTab.classList.remove('text-indigo-100', 'hover:bg-indigo-500');
        }

        /**
         * Generates the HTML for the brainstorming questions section.
         */
        function renderBrainstorming() {
            const list = document.getElementById('brainstorm-list');
            list.innerHTML = brainstormingQuestions.map((q, index) => `
                <div class="p-4 bg-gray-50 border border-gray-200 rounded-lg shadow-sm">
                    <p class="text-sm font-semibold text-indigo-600 mb-1">Question #${index + 1}</p>
                    <p class="text-gray-700 font-medium">${q}</p>
                </div>
            `).join('');
        }

        /**
         * Generates the HTML for the jokes section with a reveal mechanism.
         */
        function renderJokes() {
            const list = document.getElementById('jokes-list');
            list.innerHTML = jokes.map((joke, index) => {
                const isKnockKnock = joke.type === 'knockknock';
                return `
                    <div id="joke-${index}" class="p-4 bg-green-50 border border-green-200 rounded-lg shadow-md transition duration-300 hover:shadow-lg">
                        <p class="font-bold text-green-700">${isKnockKnock ? 'Setup: (Start of conversation)' : 'Question:'}</p>
                        <p class="text-gray-800 mb-3">${joke.setup}</p>
                        
                        <div id="punchline-${index}" class="punchline hidden p-2 bg-green-200 rounded-md mt-2 text-green-900 font-semibold border-l-4 border-green-600">
                            ${isKnockKnock ? 'Answer: (The punchline)' : 'Answer:'} ${joke.punchline}
                        </div>
                        <button onclick="revealPunchline(${index})" class="mt-3 w-full sm:w-auto text-sm font-medium text-white bg-green-600 py-2 px-4 rounded-lg hover:bg-green-700 transition duration-150 shadow-md">
                            Reveal The Funny Part!
                        </button>
                    </div>
                `;
            }).join('');
        }

        /**
         * Reveals the punchline for a specific joke.
         * @param {number} index The index of the joke to reveal.
         */
        function revealPunchline(index) {
            const punchlineElement = document.getElementById(`punchline-${index}`);
            const buttonElement = punchlineElement.nextElementSibling;
            if (punchlineElement.classList.contains('hidden')) {
                punchlineElement.classList.remove('hidden');
                buttonElement.textContent = 'Hide Punchline';
            } else {
                punchlineElement.classList.add('hidden');
                buttonElement.textContent = 'Reveal The Funny Part!';
            }
        }

        /**
         * Generates the HTML for the riddles section with a game-like reveal.
         */
        function renderRiddles() {
            const container = document.getElementById('riddle-container');
            container.innerHTML = riddlesAndPuzzles.map((riddle, index) => `
                <div id="riddle-card-${index}" class="p-5 bg-pink-50 border border-pink-200 rounded-xl shadow-md">
                    <p class="text-lg font-bold text-pink-700 mb-2">Riddle #${index + 1}</p>
                    <p class="text-gray-800 text-xl font-medium mb-4 italic">"${riddle.riddle}"</p>
                    
                    <div id="answer-${index}" class="riddle-answer hidden p-3 bg-pink-200 border-l-4 border-pink-600 rounded-md mt-4 text-pink-900 font-bold text-center">
                        Answer: ${riddle.answer}
                    </div>
                    
                    <button onclick="revealAnswer(${index})" class="mt-4 w-full font-medium text-white bg-pink-600 py-2 rounded-lg hover:bg-pink-700 transition duration-150 shadow-lg hover:shadow-xl">
                        Reveal Answer
                    </button>
                </div>
            `).join('');
        }

        /**
         * Reveals the answer for a specific riddle.
         * @param {number} index The index of the riddle to reveal.
         */
        function revealAnswer(index) {
            const answerElement = document.getElementById(`answer-${index}`);
            const buttonElement = answerElement.nextElementSibling;
            if (answerElement.classList.contains('hidden')) {
                answerElement.classList.remove('hidden');
                buttonElement.textContent = 'Hide Answer';
                buttonElement.classList.add('bg-pink-800');
                buttonElement.classList.remove('bg-pink-600');
            } else {
                answerElement.classList.add('hidden');
                buttonElement.textContent = 'Reveal Answer';
                buttonElement.classList.remove('bg-pink-800');
                buttonElement.classList.add('bg-pink-600');
            }
        }

        // --- Initialization ---

        window.onload = function() {
            // Render content for all sections
            renderBrainstorming();
            renderJokes();
            renderRiddles();

            // Set the initial view to 'brainstorm'
            showSection('brainstorm');
        };

    </script>
